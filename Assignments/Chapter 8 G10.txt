clear, clc
function p=polyadd(p1,p2,operation)
np1=length(p1);
np2=length(p2);
% Padding p2, if shorter than p1.
if np1>np2
    nd=np1-np2;
    p2add(1:nd)=0;
    p2=[p2add p2];
end
% Padding p1, if shorter than p2.
if np2>np1
    nd=np2-np1;
    p1add(1:nd)=0;
    p1=[p1add p1];
end
switch operation
    case 'add'
        p=p1+p2;
    case 'sub'
        p=p1-p2;
end
end

%16
function p = polymult(p1,p2)
%Multiply polynomials
na=length(p1); nb=length(p2);
if nb > na
    d=p1; p1=p2;
    clear b
    p2=d;
    nd=na; na=nb; nb=nd;
end
for k=1:nb
    p(k)=0;
    for i=1:k
        p(k)=p(k)+p1(i)*p2(k+1-i);
    end
end
for k=nb+1:na
    p(k)=0;
    for i=k-nb+1:k
        p(k)=p(k)+p1(i)*p2(k+1-i);
    end
end
for k=na+1:na+nb-1
    p(k)=0;
    for i=k-nb+1:na
        p(k)=p(k)+p1(i)*p2(k+1-i);
    end
end
end
